\newcommand{\faketoc}[1]{%
	\chapter{#1}%
}

\makeatletter
\newlength\toclength
\setlength{\toclength}{\@pnumwidth}
\makeatother

% ------------------
% part
% ------------------

\newcounter{fakepartCounter}[chapter]
\setcounter{fakepartCounter}{0}

\newcommand{\fakepart}[2][]{
	\refstepcounter{fakepartCounter}
	\vspace{10.0bp}
	\noindent
	{
		\hfill
		\begin{minipage}{0.60\textwidth}% ? (宽度)
			\CJKfamily{xb}%
			\fontsize{11.0bp}{11.0bp}\selectfont\ziju{0.150}%
			\centering%
			#2
		\end{minipage}
		\hfill
	}
	\vspace{10.0bp}
	\par
}

% ------------------
% chapter
% ------------------

\newcounter{fakechapterCounter}[fakepartCounter]
\setcounter{fakechapterCounter}{0}

\newcommand{\fakechapter}[2][]{
	\refstepcounter{fakechapterCounter}
	\begingroup
	\setlength{\leftskip}{-1.7em}
	\hangafter 1
	\hangindent 5.2em
	{
		{
			\CJKfamily{ss}%
			\fontsize{9.0bp}{9.0bp}\selectfont\ziju{0.010}% ? (行距)
			第\chinese{fakechapterCounter}篇\hspace{1.0em}#2
		}
    	\ifnum\pdfstrcmp{#1}{1}=0
			\titlerule*[.4em]{$\cdot$}\makebox[\toclength][r]{\pageref{#2}}
    	\fi
	}
	\par
	\endgroup
}
